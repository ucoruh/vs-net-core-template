     303             : 
     304             : Use following pre-commit script and replace it with `.git/hook/precommit`
     305             : 
     306             : ```bash
     307             : #!/bin/bash
     308             : # Installation:
     309             : #   Copy this script as pre-commit to .git/hooks folder
     310             : 
     311             : OPTIONS="--options=astyle-options.txt"
     312             : 
     313             : # Check if .gitignore file exists
     314             : if [ ! -f astyle-options.txt ]; then
     315             :     echo "[!] astyle-options.txt file not found. Please make sure the astyle-options.txt file is present." >&2
     316             :     exit 1
     317             : fi
     318             : 
     319             : RETURN=0
     320             : ASTYLE=$(command -v astyle)
     321             : if [ $? -ne 0 ]; then
     322             :     echo "[!] AStyle is not installed. Unable to check source file format policy." >&2
     323             :     exit 1
     324             : fi
     325             : 
     326             : # Check if .gitignore file exists
     327             : if [ ! -f .gitignore ]; then
     328             :     echo "[!] .gitignore file not found. Please make sure the .gitignore file is present." >&2
     329             :     exit 1
     330             : fi
     331             : 
     332             : # Check if README.md file exists
     333             : if [ ! -f README.md ]; then
     334             :     echo "[!] README.md file not found. Please make sure the README.md file is present." >&2
     335             :     exit 1
     336             : fi
     337             : 
     338             : # Check if .gitignore file exists
     339             : if [ ! -f Doxyfile ]; then
     340             :     echo "[!] Doxyfile file not found. Please make sure the Doxyfile file is present." >&2
     341             :     exit 1
     342             : fi
     343             : 
     344             : FILES=$(git diff --cached --name-only --diff-filter=ACMRTUXB | grep -E "\.(cs|java|c|cpp|h)$")
     345             : for FILE in $FILES; do
     346             :     FORMATTED=$( "$ASTYLE" $OPTIONS < "$FILE" 2>&1 )
     347             :     if [ $? -ne 0 ]; then
     348             :         echo "[!] Error formatting $FILE:" >&2
     349             :         echo "$FORMATTED" >&2
     350             :         RETURN=1
     351             :     elif [ -n "$FORMATTED" ]; then
     352             :         echo "Formatted $FILE"
     353             :         echo "$FORMATTED"
     354             :         echo ""
     355             :         echo "$FORMATTED" > "$FILE"
     356             :         git add "$FILE"
     357             :     fi
     358             : done
     359             : 
     360             : if [ $RETURN -eq 1 ]; then
     361             :     echo "" >&2
     362             :     echo "Make sure you have run AStyle with the following options:" >&2
     363             :     echo $OPTIONS >&2
     364             : fi
     365             : 
     366             : exit $RETURN
     367             : ```

     369             : This script will check astyle-options.txt file, gitignore file, README.md file and Doxyfile is exist if not it will be terminate your commit operation. Formatting configuration will be stored in `astyle-options.txt` .
